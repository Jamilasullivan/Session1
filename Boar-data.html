Making predictions

unique(boar$Tb) # look at Tb levels for y
## [1]  0  1 NA
pdat <- expand.grid(length = seq(46,150,0.1), age = 1,2,3,4, sex = 1,2) # Tb is the y variable so we are giving R a fake scenario and asking it to make new y values by providing all of the x values in a dataframe 

pred <- predict (mod7, newdata = pdat, na.rm = T,
                 type= "response", se.fit = TRUE) # predication of the risk of getting Tb as a function of length 

predframe <- data.frame (pdat, preds = pred$fit, se = pred$se.fit)

par (mfrow = c(1,1))
plot(Tb~length, data = boar)
str(boar)
